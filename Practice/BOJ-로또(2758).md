```c++
/*
DP. BOTTOM-UP
*/
using namespace std;

int n, m;
long long dp[11][2001];
int main() {

   int T;
   cin >> T;

   for (int i = 1; i <= 2000; i++) 
      dp[1][i] = i;
   for (int i = 2; i <= 10; i++) {
      for (int j = 1; j <= 2000; j++) {
         dp[i][j] += dp[i - 1][j / 2] + dp[i][j - 1];
      }
   }

   for (int tc = 0; tc < T; tc++) {
      
      cin >> n >> m;
      cout << dp[n][m] << endl;

   }


   return 0;
}
```
