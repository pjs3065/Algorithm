```c++
//BFS, 완탐
#include <iostream>
#include <cstdio>
#include <queue>
#include <utility>

using namespace std;

int n, m;
int map[1001][1001];
bool chk[1001][1001];

int dx[] = { 1,0,-1,0 };
int dy[] = { 0,1,0,-1 };
void solution() {
	
	queue<pair<pair<int, int>, int>>  que;

	que.push(make_pair(make_pair(0, 0), 0));
	chk[0][0] = true;

	map[0][0] = 1;

	while (!que.empty()) {
		int queSize = que.size();
		for (int z = 0; z < queSize; z++) {
			int cx = que.front().first.first;
			int cy = que.front().first.second;
			int cWall = que.front().second;
			que.pop();
			for (int i = 0; i < 4; i++) {
				int nx = cx + dx[i];
				int ny = cy + dy[i];
				
				if (nx < 0 || nx >= n || ny < 0 || ny >= m || chk[nx][ny])
					continue;

				if (map[nx][ny] == 0) {
					chk[nx][ny] = true;
					map[nx][ny] = map[cx][cy] + 1;
					que.push(make_pair(make_pair(nx, ny), cWall));
				}
				if ((map[nx][ny] == 1 && cWall == 0)) {
					chk[nx][ny] = true;
					map[nx][ny] = map[cx][cy] + 1;
					que.push(make_pair(make_pair(nx, ny), 1));
				}
			}
		}
	}
	
}
int main() {

	scanf("%d %d", &n, &m);

	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%1d", &map[i][j]);
		}
	}

	solution();

	printf("%d\n", map[n - 1][m - 1]);

	return 0;
}
```
