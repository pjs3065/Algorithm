```c++
//bfs, 큐 2개 사용, 턴 문제
#include <iostream>
#include <queue>
#include <utility>
#include <vector>
#include <cstring>
using namespace std;

int w, h;
char map[1001][1001];
bool chk[1001][1001];
bool chk2[1001][1001];
pair<int, int> player;
vector<pair<int, int>> fire;
pair<int, int> prevLoc;

int dx[] = { 1, 0, -1, 0 };
int dy[] = { 0, 1, 0, -1 };
void solution() {

	for (int i = 0; i < h; i++) {
		memset(chk[i], false, sizeof(chk[i]));
		memset(chk2[i], false, sizeof(chk2[i]));
	}

	queue<pair<int, int>> fque, pque;

	for (int i = 0; i < fire.size(); i++) {
		fque.push(fire[i]);
		chk2[fire[i].first][fire[i].second] = true;
	}

	pque.push(player);
	chk[player.first][player.second] = true;

	int time = 0;

	while (!pque.empty()) {

		int queSize_fire = fque.size();
		int queSize_player = pque.size();
		time++;

		for (int z = 0; z < queSize_fire; z++) {
			int fcx = fque.front().first;
			int fcy = fque.front().second;
			fque.pop();
			for (int i = 0; i < 4; i++) {
				int fnx = fcx + dx[i];
				int fny = fcy + dy[i];
				if (fnx >= 0 && fnx < h && fny >= 0 && fny < w) {
					if (!chk2[fnx][fny] && map[fnx][fny] != '#') {
						map[fnx][fny] = '*';
						chk2[fnx][fny] = true;
						fque.push(make_pair(fnx, fny));
					}
				}
			}
		}
		for (int z = 0; z < queSize_player; z++) {
			int pcx = pque.front().first;
			int pcy = pque.front().second;
			if (pcx == 0 || pcx == h - 1
				|| pcy == 0 || pcy == w - 1) {
				cout << time << endl;
				return;
			}
			pque.pop();
			for (int i = 0; i < 4; i++) {
				int pnx = pcx + dx[i];
				int pny = pcy + dy[i];

				if (pnx >= 0 && pnx < h && pny >= 0 && pny < w) {
					if (map[pnx][pny] == '@')map[pcx][pcy] = '.';
					if (!chk[pnx][pny] && map[pnx][pny] == '.') {
						if (map[pcx][pcy] != '*')
							map[pcx][pcy] = '.';

						map[pnx][pny] = '@';
						chk[pnx][pny] = true;
						pque.push(make_pair(pnx, pny));
					}
				}
			}
		}
	}

	cout << "IMPOSSIBLE" << endl;

}
int main() {
	int T;
	cin >> T;
	for (int tc = 0; tc < T; tc++) {
		cin >> w >> h;
		fire.clear();
		for (int i = 0; i < h; i++) {
			for (int j = 0; j < w; j++) {
				cin >> map[i][j];
				if (map[i][j] == '*') {
					fire.push_back(make_pair(i, j));
				}
				if (map[i][j] == '@') {
					player.first = i; player.second = j;
				}
			}
		}
		solution();
	}

	return 0;
}
```
