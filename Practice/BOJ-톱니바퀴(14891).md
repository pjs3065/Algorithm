```c++
#include <cstdio>
#include <algorithm>
#include <utility>
#include <vector>
#include <cstring>
using namespace std;

int a[4][8];
vector<pair<int, int>> cmd;
int k;
void solution() {
	
	for (int i = 0; i < k; i++) {
		int idx = cmd[i].first;
		int dir = cmd[i].second;

		int leftIdx = idx - 1;
		int leftDir = dir == 1 ? -1 : 1;
	
		int rightIdx = idx + 1;
		int rightDir = dir == 1 ? -1 : 1;
		
		int leftCount = 0;
		int rightCount = 0;
		
		//왼쪽 방향 회전하는 톱니바퀴 개수 저장
		while (leftIdx >= 0) {
			if (a[leftIdx + 1][6] == a[leftIdx][2]) 
				break;
			else {
				leftCount++;
				leftIdx--;
			}
		}

		leftIdx = idx - 1;
		//톱니바퀴 회전
		while (leftCount > 0) {
				if (leftDir == 1) { //시계방향일 경우,
					int tmp = a[leftIdx][7];
					memmove(a[leftIdx] + 1, a[leftIdx], sizeof(int) * 7); //(복사 시작할 주소, 복사 시작 주소, 복사시작에서 몇개)
					a[leftIdx][0] = tmp;
					leftIdx--;
					leftDir = -1;
				}
				else { //반 시계방향일 경우,
					int tmp = a[leftIdx][0];
					memmove(a[leftIdx], a[leftIdx] + 1, sizeof(int) * 7); //
					a[leftIdx][7] = tmp;
					leftIdx--;
					leftDir = 1;
				}
				leftCount--;
		}


		//오른쪽 방향 회전하는 톱니바퀴 개수 저장
		while (rightIdx < 4) {
			if (a[rightIdx-1][2] == a[rightIdx][6])
				break;
			else {
				rightCount++;
				rightIdx++;
			}
		}

		rightIdx = idx + 1;
		//톱니바퀴 회전
		while (rightCount > 0) {
			if (rightDir == 1) { //시계방향일 경우,
				int tmp = a[rightIdx][7];
				memmove(a[rightIdx] + 1, a[rightIdx], sizeof(int) * 7); //(복사 될 주소, 복사 시작 할 주소, 복사시작에서 몇개)
				a[rightIdx][0] = tmp;
				rightIdx++;
				rightDir = -1;
			}
			else { //반 시계방향일 경우,
				int tmp = a[rightIdx][0];
				memmove(a[rightIdx], a[rightIdx] + 1, sizeof(int) * 7); //
				a[rightIdx][7] = tmp;
				rightIdx++;
				rightDir = 1;
			}
			rightCount--;
		}

		//현재 톱니바퀴 위치 조정
		if (dir == 1) {
			int tmp = a[idx][7];
			memmove(a[idx] + 1, a[idx], sizeof(int) * 7);
			a[idx][0] = tmp;
		}
		else {
			int tmp = a[idx][0];
			memmove(a[idx] , a[idx] + 1, sizeof(int) * 7);
			a[idx][7] = tmp;
		}
	}


	int answer = a[0][0]+ (a[1][0] * 2) + (a[2][0] * 4) + (a[3][0] * 8);
	printf("%d\n", answer);

}
int main() {

	for (int i = 0; i < 4; i++) {
		for (int j = 0; j < 8; j++) {
			scanf("%1d", &a[i][j]);
		}
	}
	
	scanf("%d",&k);
	for (int i = 0; i < k; i++) {
		int idx, dir;
		scanf("%d %d", &idx, &dir);
		cmd.push_back(make_pair(idx - 1, dir));
	}

	solution();
	
	return 0;
}
```
